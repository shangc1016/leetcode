### 数组的旋转操作

```cpp
// 顺时针旋转90
for (int i = 1; i <= size; i++) {
    for (int j = 1; j <= size; j++) {
        after[i][j] = before[size - j + 1][i];
    }
}


// 顺时针旋转270、逆时针旋转90
for (int i = 1; i <= size; i++) {
    for (int j = 1; j <= size; j++) {
        result[i][j] = first[j][size - i + 1];
    }
}


// 反射，水平翻转
for (int i = 1; i <= size; i++) {
    for (int j = 1; j <= size; j++) {
        result[i][j] = first[i][size - j + 1];
    }
}

// 反射，上下翻转
for (int i = 1; i <= size; i++) {
    for (int j = 1; j <= size; j++) {
        result[i][j] = first[size - i + 1][j];
    }
}

```
